```swift swift
var request = URLRequest(url: URL(string: "https://www.themealdb.com/api/json/v1/1/search.php")!,timeoutInterval: Double.infinity)

let task = URLSession.shared.dataTask(with: request) { data, response, error in

    print(String(describing: error))
    return
  }
  print(String(data: data, encoding: .utf8)!)
}

task.resume()
```
